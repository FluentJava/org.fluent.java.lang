001package tm.lang;
002
003import static org.junit.Assert.*;
004
005import org.junit.Before;
006import org.junit.Test;
007
008public class Reflection_Test 
009{
010        Reflection reflection;
011        TestObject testObject;
012        
013        public Reflection_Test()
014        {
015                testObject = new TestObject();
016                reflection = new Reflection(testObject);
017        }
018        
019        @Test 
020        public void Reflection_Ctor()
021        {                               
022                assertNotNull(reflection);
023                assertNotNull(reflection.target);
024                assertNotNull(reflection.clazz);
025                assertNotNull(reflection.methods);
026                assertNotNull(reflection.methods.reflection);
027                assertEquals (reflection.clazz                 , TestObject.class);
028                assertEquals (reflection.target                , testObject);           
029                assertEquals (reflection.methods.reflection, reflection);
030        }
031}
