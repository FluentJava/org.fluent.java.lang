001package tm.lang;
002
003import static org.junit.Assert.*;
004
005import java.lang.reflect.InvocationTargetException;
006
007import org.junit.Test;
008
009import tm.lang.reflect.Method;
010
011public class Method_Test 
012{
013        public TestObject testObject;
014        public Reflection reflection ;          
015        
016        public Method_Test() 
017        {
018                testObject = new TestObject();
019                reflection = new Reflection(testObject);                 
020        }
021        
022        @Test 
023        public void Method_Ctor()
024        {               
025                Method method = reflection.method("static_Method");
026                assertNotNull(method);
027                assertNotNull(method.method_Obj);
028                assertNotNull(method.reflection);
029                assertNull   (method.lastInvocationException);
030                assertFalse  (method.logInvocationExeption);
031                assertEquals (method.reflection, reflection);
032        }
033        
034        @Test 
035        public void isStatic()
036        {
037                Method static_Method = reflection.method("static_Method");
038                assertTrue (static_Method.isStatic  ());
039                assertFalse(static_Method.isInstance());
040                
041                Method instance_Method = reflection.method("instance_Method");
042                assertFalse(instance_Method.isStatic  ());
043                assertTrue (instance_Method.isInstance());
044        }
045        @Test 
046        public void isInstance()
047        {
048                //covered by: isStatic()
049        }
050        @Test 
051        public void invoke()
052        {               
053                String echoValue = "Hello World"; 
054                //instance invocation
055                Method instance_Method = reflection.method("instance_Method");
056                Method instance_Echo   = reflection.method("instance_Echo");
057                
058                assertNotNull(instance_Method);
059                assertNotNull(instance_Echo);
060                assertEquals (instance_Method.invoke()             , TestObject.instance_Return);
061                assertEquals (instance_Echo.  invoke(echoValue), echoValue);
062                
063                //static invocation
064                Method static_Method = reflection.method("static_Method");
065                Method static_Echo   = reflection.method("static_Echo");
066                
067                assertNotNull(static_Method);
068                assertNotNull(static_Echo);
069                assertEquals (static_Method.invoke()             , TestObject.static_Return);
070                assertEquals (static_Echo.  invoke(echoValue), echoValue);  
071                
072                //test nulls and exception handing
073                assertNull(new Method(null,null).invoke());
074                assertNull(static_Echo.invoke(0));                      // static_Echo parameter is String
075                static_Echo.logInvocationExeption = true;       // ideally we should capture the log to see if it did really happened (and the moment the logs go to System.out)
076                assertNull(static_Echo.invoke(0));                      // static_Echo parameter is String
077        }
078        
079        @Test
080        public void invoke_No_Catch() throws Exception
081        {       
082                //instance invocation
083                try
084                {
085                        Method instance_Echo = reflection.method("instance_Echo");                      
086                        assertNotNull(instance_Echo);
087                        instance_Echo.invoke_No_Catch(0);                       // instance_Echo parameter is String
088                        fail("instance_Echo.invoke_No_Catch should had thrown exception");
089                }
090                catch(Exception ex) {}
091                //static invocation
092                try
093                {
094                        Method static_Echo = reflection.method("static_Echo");                  
095                        assertNotNull(static_Echo);
096                        static_Echo.invoke_No_Catch(0);                 // static_Echo parameter is String
097                        fail("static_Echo.invoke_No_Catch should had thrown exception");
098                }
099                catch(Exception ex) {}
100                
101                //test nulls
102                assertNull(new Method(null,null).invoke_No_Catch());
103        }
104        
105        @Test
106        public void name()
107        {
108                String name = "static_Method";
109                Method method = reflection.method(name);
110                assertEquals(method.name(), name);
111                assertTrue  (method.name(name));
112                assertFalse (method.name(null));
113                assertFalse (method.name(name + "AA"));
114                assertNull  (new Method(null,null).name());
115                assertFalse (new Method(null,null).name(null));
116        }
117        @Test 
118        public void _toString()
119        {
120                String name = "static_Method";
121                Method method = reflection.method(name);
122                assertEquals(method.toString(),   name);
123                assertNull  (new Method(null,null).toString());
124        }
125}
