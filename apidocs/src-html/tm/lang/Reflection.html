001package tm.lang;
002
003import java.lang.reflect.Field;
004import java.lang.reflect.InvocationTargetException;
005import java.util.ArrayList;
006import java.util.Arrays;
007import java.util.Collections;
008import java.util.Comparator;
009import java.util.List;
010
011import tm.lang.reflect.Method;
012import tm.lang.reflect.Methods;
013
014public class Reflection  
015{
016        public Object            target;
017        public Class<?>    clazz; 
018        public Methods           methods;
019        
020        public Reflection(Object target) 
021        {
022                 this(target, target.getClass());
023        }
024        
025        public Reflection(Object target, Class<?> clazz)
026        {
027                this.target  = target;
028                this.clazz   = clazz;
029                this.methods = new Methods      (this);
030        }
031         
032        public static Reflection reflection(Object target)
033        {
034                Reflection reflection = new Reflection(target);         
035                return reflection;
036        }
037                
038        public Method method(String methodName)
039        {
040                return methods.method(methodName);
041        }
042        
043        
044        public Field field(String fieldName)
045        {
046                
047                try 
048                {
049                        return clazz.getDeclaredField(fieldName);
050                } 
051                catch (SecurityException    e)  {       e.printStackTrace(); } 
052                catch (NoSuchFieldException e)  {       e.printStackTrace(); }
053                return null;
054        }
055        public Object field_Value(String fieldName)
056        {               
057                return field_Value(field(fieldName));
058        }
059        public <T> T field_Value(String fieldName, Class<T> returnType)
060        {
061                Object value = field_Value(fieldName);
062                if (value.getClass().equals(returnType))
063                        return returnType.cast(value);
064                return null;
065        }
066        public Object field_Value(Field field)
067        {
068                if (field!= null)
069                {
070                        field.setAccessible(true);
071                        try 
072                        {
073                                return field.get(target);
074                        } 
075                        catch (IllegalArgumentException e) 
076                        {
077                                e.printStackTrace();
078                        } catch (IllegalAccessException e) 
079                        {
080                                e.printStackTrace();
081                        }
082                }
083                return null;
084        }
085        public List<Field> fields()
086        {
087                List<Field> fields = Arrays.asList(clazz.getDeclaredFields()); 
088                return sort_by_Name(fields);
089        }
090        public List<Object> fields_Values()
091        {
092                return fields_Values(fields());
093        }
094        public List<Object> fields_Values(List<Field> fields)
095        {
096                List<Object> values = new ArrayList<Object>();
097                for(Field field : fields)
098                        values.add(field_Value(field));
099                return values;
100        }
101        
102        public List<String> fields_Names()
103        {
104                List<String> names = new ArrayList<String>();
105                for(Field field : fields())                     
106                        names.add(field.getName());
107                return names;   
108        }
109
110        public List<String> getters_Names()
111        {
112                List<String> getters = new ArrayList<String>();
113                for(String methodName : methods.methods_Names())
114                        if (methodName.startsWith("get"))
115                                        getters.add(methodName.substring(3));
116                return getters;
117        }
118        public List<String> getters_Values()
119        {
120                List<String> getters = new ArrayList<String>();
121                for(String methodName : methods.methods_Names())
122                        if (methodName.startsWith("get"))
123                        {
124                                Object result = methods.invoke(methodName);                             
125                                getters.add(result == null ? null : result.toString());                                 
126                        }
127                return getters;
128        }       
129        
130        
131        //Utils
132        
133        public List<String> sort(List<String> list)
134        {
135                Collections.sort(list);
136                return list;
137        }
138        //TODO this is throwing a lot of fieldNameNotFound exceptions
139        public <T> List<T> sort_by_Name(List<T> list)
140        {               
141                //Collections.sort(list, new Comparator_FieldToStringValue<T>("name"));
142                Collections.sort(list, new Comparator_FieldToStringValue<T>("name"));
143                return list;            
144        }       
145        
146        public Reflection superClass()
147        {
148                Class<?> superClass  = clazz.getSuperclass();     
149                return  (superClass != null) 
150                                        ? new Reflection(this, superClass)
151                                        : null;
152        }       
153
154        
155        //TODO move to another class
156        public static class Comparator_FieldToStringValue<T> implements Comparator<T>
157        {
158                public String fieldToSortBy;
159                public Comparator_FieldToStringValue(String fieldToSortBy)
160                {
161                        this.fieldToSortBy = fieldToSortBy;
162                }
163        
164                public int compare(T object1, T object2)
165                {
166                        try
167                        {
168                                String value1 = new Reflection(object1).field_Value(fieldToSortBy).toString();
169                                String value2 = new Reflection(object2).field_Value(fieldToSortBy).toString();
170                                return value1.compareTo(value2);
171                        }
172                        catch(Exception ex)
173                        {
174                                return 0;
175                        }
176                }
177        }
178}
